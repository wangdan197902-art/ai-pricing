## 1. 架构总览

### 1.1 架构原则

| 原则 | 说明 |
|------|------|
| 静态优先 | 全站 Hugo 静态生成，零服务器、零数据库、零冷启动 |
| 数据驱动 | 所有工具信息以 JSON 数据文件为唯一事实源，页面从数据生成 |
| AI 自动化 | 内容生产、翻译、配图、审核全部由 AI 完成，人工仅审核 |
| 多语种原生 | Hugo 多语言站，10 语种独立 URL，每页面 ×10 镜像 |
| 成本最优 | 全栈 Serverless + 免费额度，年度硬成本 $10-15 |
| SEO/AEO 双优 | 同时面向搜索引擎(Product JSON-LD)和 AI 引擎(llms.txt) |
| 可观测 | GitHub Actions + Cloudflare Analytics + UptimeRobot 全链路监控 |

### 2.2 代码组织原则

- **数据/内容/模板三分离**: `data/` 是事实源,`content/` 是渲染内容,`layouts/` 是表现层
- **语种平行**: `content/{lang}/` 结构完全一致,翻译仅替换文本
- **脚本复用**: `scripts/lib/` 集中放工具函数,主脚本仅编排流程
- **配置分级**: `config/_default` 基础,`config/production` 覆盖生产参数

### 5.2 AI 客户端封装 `scripts/lib/ai_client.py`

```python
"""
AI 客户端统一封装: Claude / GPT-4o / Gemini
所有调用走此模块,便于限流、缓存、错误重试。
"""
import os
import json
import time
import hashlib
from pathlib import Path
from typing import Optional

import requests

CACHE_DIR = Path(".cache/ai")
CACHE_DIR.mkdir(parents=True, exist_ok=True)


class AIClient:
    """统一 AI 客户端基类。"""

    def __init__(self, provider: str, api_key: str, model: str, max_retries: int = 3):
        self.provider = provider
        self.api_key = api_key
        self.model = model
        self.max_retries = max_retries

    def _cache_key(self, prompt: str, **kwargs) -> str:
        payload = json.dumps({"prompt": prompt, **kwargs}, sort_keys=True)
        return hashlib.sha256(payload.encode()).hexdigest()

    def _cached(self, key: str) -> Optional[str]:
        path = CACHE_DIR / f"{self.provider}_{key}.txt"
        if path.exists():
            return path.read_text(encoding="utf-8")
        return None

    def _save_cache(self, key: str, value: str) -> None:
        path = CACHE_DIR / f"{self.provider}_{key}.txt"
        path.write_text(value, encoding="utf-8")

    def _retry(self, fn, *args, **kwargs):
        last_exc = None
        for attempt in range(self.max_retries):
            try:
                return fn(*args, **kwargs)
            except Exception as e:
                last_exc = e
                wait = 2 ** attempt
                print(f"[retry] {self.provider} attempt {attempt + 1} failed: {e}, wait {wait}s")
                time.sleep(wait)
        raise last_exc


class ClaudeClient(AIClient):
    """Anthropic Claude 客户端。"""

    ENDPOINT = "https://api.anthropic.com/v1/messages"
    ANTHROPIC_VERSION = "2023-06-01"

    def __init__(self, api_key: str = None, model: str = "claude-3-5-sonnet-20241022"):
        super().__init__("claude", api_key or os.getenv("ANTHROPIC_API_KEY"), model)

    def complete(self, prompt: str, system: str = "", max_tokens: int = 4096, temperature: float = 0.7, use_cache: bool = True) -> str:
        if use_cache:
            cached = self._cached(self._cache_key(prompt, system=system, model=self.model, max_tokens=max_tokens, temperature=temperature))
            if cached:
                print("[claude] cache hit")
                return cached

        def call():
            resp = requests.post(
                self.ENDPOINT,
                headers={
                    "x-api-key": self.api_key,
                    "anthropic-version": self.ANTHROPIC_VERSION,
                    "content-type": "application/json",
                },
                json={
                    "model": self.model,
                    "max_tokens": max_tokens,
                    "temperature": temperature,
                    "system": system,
                    "messages": [{"role": "user", "content": prompt}],
                },
                timeout=120,
            )
            resp.raise_for_status()
            data = resp.json()
            return data["content"][0]["text"]

        result = self._retry(call)
        if use_cache:
            self._save_cache(self._cache_key(prompt, system=system, model=self.model, max_tokens=max_tokens, temperature=temperature), result)
        return result


class GPT4oClient(AIClient):
    """OpenAI GPT-4o 客户端。"""

    ENDPOINT = "https://api.openai.com/v1/chat/completions"

    def __init__(self, api_key: str = None, model: str = "gpt-4o"):
        super().__init__("gpt4o", api_key or os.getenv("OPENAI_API_KEY"), model)

    def complete(self, prompt: str, system: str = "", max_tokens: int = 4096, temperature: float = 0.3, use_cache: bool = True) -> str:
        if use_cache:
            cached = self._cached(self._cache_key(prompt, system=system, model=self.model, max_tokens=max_tokens, temperature=temperature))
            if cached:
                return cached

        def call():
            resp = requests.post(
                self.ENDPOINT,
                headers={"Authorization": f"Bearer {self.api_key}", "Content-Type": "application/json"},
                json={
                    "model": self.model,
                    "messages": [{"role": "system", "content": system}, {"role": "user", "content": prompt}],
                    "max_tokens": max_tokens,
                    "temperature": temperature,
                },
                timeout=120,
            )
            resp.raise_for_status()
            return resp.json()["choices"][0]["message"]["content"]

        result = self._retry(call)
        if use_cache:
            self._save_cache(self._cache_key(prompt, system=system, model=self.model, max_tokens=max_tokens, temperature=temperature), result)
        return result


class GeminiClient(AIClient):
    """Google Gemini 客户端(用于图像生成与多模态)。"""

    ENDPOINT = "https://generativelanguage.googleapis.com/v1beta/models"

    def __init__(self, api_key: str = None, model: str = "gemini-1.5-flash"):
        super().__init__("gemini", api_key or os.getenv("GEMINI_API_KEY"), model)

    def complete(self, prompt: str, max_tokens: int = 4096, temperature: float = 0.7, use_cache: bool = True) -> str:
        if use_cache:
            cached = self._cached(self._cache_key(prompt, model=self.model, max_tokens=max_tokens, temperature=temperature))
            if cached:
                return cached

        def call():
            url = f"{self.ENDPOINT}/{self.model}:generateContent?key={self.api_key}"
            resp = requests.post(
                url,
                json={
                    "contents": [{"parts": [{"text": prompt}]}],
                    "generationConfig": {"maxOutputTokens": max_tokens, "temperature": temperature},
                },
                timeout=120,
            )
            resp.raise_for_status()
            return resp.json()["candidates"][0]["content"]["parts"][0]["text"]

        result = self._retry(call)
        if use_cache:
            self._save_cache(self._cache_key(prompt, model=self.model, max_tokens=max_tokens, temperature=temperature), result)
        return result
```

### 5.3 Schema 校验 `scripts/lib/schema.py`

```python
"""JSON Schema 校验工具,在数据入库前强制校验。"""
import json
from pathlib import Path
from typing import List, Tuple

try:
    import jsonschema
except ImportError:
    jsonschema = None


def validate(data: dict, schema_path: str) -> Tuple[bool, List[str]]:
    if jsonschema is None:
        return True, ["jsonschema not installed, skip"]
    schema = json.loads(Path(schema_path).read_text(encoding="utf-8"))
    validator = jsonschema.Draft7Validator(schema)
    errors = sorted(validator.iter_errors(data), key=lambda e: e.path)
    if not errors:
        return True, []
    return False, [f"{'.'.join(map(str, e.path))}: {e.message}" for e in errors]


def validate_tool(tool: dict, project_root: str = ".") -> Tuple[bool, List[str]]:
    return validate(tool, f"{project_root}/data/schema/tool.schema.json")


def validate_price(price: dict, project_root: str = ".") -> Tuple[bool, List[str]]:
    return validate(price, f"{project_root}/data/schema/price.schema.json")
```

```python
import subprocess
import sys

STEPS = [
    ("fetch_tools", "python3 scripts/fetch_tools.py"),
    ("fetch_prices", "python3 scripts/fetch_prices.py --with-diff"),
    ("generate_comparison", "python3 scripts/generate_comparison.py --all"),

    ("audit_content", "python3 scripts/audit_content.py --all"),
    ("validate_project", "python3 scripts/validate_project.py ."),
]


def run_pipeline(skip: list = None):
    skip = skip or []
    for name, cmd in STEPS:
        if name in skip:
            print(f"[skip] {name}")
            continue
        print(f"\n=== {name} ===")
        result = subprocess.run(cmd, shell=True)
        if result.returncode != 0:
            print(f"[FAIL] {name} exited {result.returncode}")
            sys.exit(1)
    print("\n✅ Pipeline complete")


if __name__ == "__main__":
    skip = sys.argv[1].split(",") if len(sys.argv) > 1 else []
    run_pipeline(skip)
```

### 7.4 llms.txt 生成 `scripts/generate_llms_txt.py`

```python
"""
生成 llms.txt — 为 AI 引擎(Claude/GPT/Perplexity/Gemini)提供结构化索引。
参考规范: https://llmstxt.org
"""
import argparse
import json
from pathlib import Path
from datetime import datetime


def generate(project_root: str, out: str):
    root = Path(project_root)
    tools_dir = root / "data" / "tools"
    lines = []
    lines.append("# AI Tools Pricing")
    lines.append("")
    lines.append("> Compare prices of 500+ AI tools. ChatGPT, Claude, Gemini, Midjourney pricing, features, and money-saving guides.")
    lines.append("")
    lines.append(f"Last updated: {datetime.utcnow().strftime('%Y-%m-%d')}")
    lines.append("")

    # 顶级页面
    lines.append("## Core Pages")
    lines.append("")
    lines.append("- [Home](https://aitoolspricing.com/en/): AI tools pricing comparison home")
    lines.append("- [All AI Tools](https://aitoolspricing.com/en/tools/): Browse 500+ AI tools by category")
    lines.append("- [Comparisons](https://aitoolspricing.com/en/comparison/): Side-by-side price and feature comparison")
    lines.append("- [Saving Guides](https://aitoolspricing.com/en/guide/): How to save on AI subscriptions")
    lines.append("- [Best AI Tools 2026](https://aitoolspricing.com/en/best-ai-tools-2026/): Curated ranking")
    lines.append("- [Pricing Calculator](https://aitoolspricing.com/en/calculator/): Estimate monthly cost")
    lines.append("")

    # 工具列表(按分类)
    lines.append("## AI Tools by Category")
    lines.append("")
    categories = {}
    for tool_file in sorted(tools_dir.glob("*.json")):
        tool = json.loads(tool_file.read_text(encoding="utf-8"))
        categories.setdefault(tool["category"], []).append(tool)

    for cat, tools in sorted(categories.items()):
        lines.append(f"### {cat.title()}")
        lines.append("")
        for t in tools:
            url = f"https://aitoolspricing.com/en/tool/{t['slug']}/pricing/"
            price = t["pricing"]["plans"][0]["price"] if t["pricing"]["plans"] else "N/A"
            lines.append(f"- [{t['name']}]({url}): from ${price}/mo — {t.get('category','')}")
        lines.append("")

    # 价格变更日志
    lines.append("## Recent Price Changes")
    lines.append("")
    prices_dir = root / "data" / "prices"
    changes = []
    for pf in prices_dir.glob("*.json"):
        ph = json.loads(pf.read_text(encoding="utf-8"))
        if ph["history"]:
            last = ph["history"][-1]
            if last.get("change_type") in ("increase", "decrease", "new_plan"):
                changes.append((last["date"], ph["tool_id"], last))

    for date, tid, h in sorted(changes, reverse=True)[:20]:
        lines.append(f"- {date}: {tid} — {h['change_type']} ${h['price']}/{h.get('billing_cycle','')}")
    lines.append("")

    Path(out).write_text("\n".join(lines), encoding="utf-8")
    print(f"✅ llms.txt written: {out} ({len(lines)} lines)")


if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--out", default="public/llms.txt")
    p.add_argument("--root", default=".")
    generate(p.parse_args().root, p.parse_args().out)
```

```python
"""部署/价格变更通知。"""
import argparse
import os
import json
from datetime import datetime
import requests


def slack_notify(webhook: str, text: str):
    if not webhook:
        print("[notify] no webhook, skip")
        return
    resp = requests.post(webhook, json={"text": text, "mrkdwn": True}, timeout=10)
    resp.raise_for_status()


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--status", choices=["success", "failure"])
    ap.add_argument("--event", choices=["price-change", "build-fail"])
    ap.add_argument("--commit", default="")
    args = ap.parse_args()
    webhook = os.getenv("SLACK_WEBHOOK", "")
    icon = "✅" if args.status == "success" or args.event == "price-change" else "❌"
    msg = f"{icon} *AI Pricing Site*\nEvent: {args.status or args.event}\nTime: {datetime.utcnow().isoformat()}\nCommit: {args.commit}"
    slack_notify(webhook, msg)


if __name__ == "__main__":
```

### 12.6 验证脚本 `scripts/validate_project.py`

```python
"""项目结构与数据完整性校验。"""
import argparse
import json
import sys
from pathlib import Path
from scripts.lib.schema import validate_tool, validate_price


REQUIRED_DIRS = [
    "archetypes", "assets", "config/_default", "content", "data",
    "data/tools", "data/prices", "data/schema", "i18n", "layouts",
    "layouts/partials", "scripts", "static", ".github/workflows",
]


def check_dirs(root: Path) -> list:
    errors = []
    for d in REQUIRED_DIRS:
        if not (root / d).is_dir():
            errors.append(f"❌ Missing dir: {d}")
    return errors


def check_tools(root: Path) -> list:
    errors = []
    schema_path = str(root / "data/schema/tool.schema.json")
    for tf in (root / "data/tools").glob("*.json"):
        try:
            data = json.loads(tf.read_text(encoding="utf-8"))
            ok, errs = validate_tool(data, str(root))
            if not ok:
                for e in errs:
                    errors.append(f"❌ {tf.name}: {e}")
            else:
                print(f"✅ {tf.name}")
        except Exception as e:
            errors.append(f"❌ {tf.name}: {e}")
    return errors


def check_prices(root: Path) -> list:
    errors = []
    for pf in (root / "data/prices").glob("*.json"):
        try:
            data = json.loads(pf.read_text(encoding="utf-8"))
            ok, errs = validate_price(data, str(root))
            if not ok:
                for e in errs:
                    errors.append(f"❌ {pf.name}: {e}")
        except Exception as e:
            errors.append(f"❌ {pf.name}: {e}")
    return errors


def check_content_locales(root: Path) -> list:
    errors = []
    locales = ["en", "zh", "ja", "ko", "es", "fr", "de", "pt", "ru", "ar"]
    for loc in locales:
        if not (root / "content" / loc).is_dir():
            errors.append(f"⚠️  Missing locale: content/{loc}")
    return errors


def check_url_files(root: Path) -> list:
    """检查图片目录是否有对应 .url 文件(本站不一定需要,但保留)。"""
    return []


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("root", default=".")
    args = ap.parse_args()
    root = Path(args.root).resolve()

    print(f"\n=== Validating project: {root} ===\n")
    errors = []
    errors += check_dirs(root)
    errors += check_tools(root)
    errors += check_prices(root)
    errors += check_content_locales(root)
    errors += check_url_files(root)

    print("\n=== Summary ===")
    if errors:
        for e in errors:
            print(e)
        print(f"\n❌ {len(errors)} errors")
        sys.exit(1)
    print("✅ All checks passed")


if __name__ == "__main__":
    main()
```
