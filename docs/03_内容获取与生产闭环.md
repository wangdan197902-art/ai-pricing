```python
"""
AI工具清单采集脚本。
从多个AI工具目录采集工具信息，去重后入库。

用法:
    python3 scripts/fetch_tools.py                     # 全量
    python3 scripts/fetch_tools.py --limit 20          # 限20个
    python3 scripts/fetch_tools.py --new-only          # 仅新增
    python3 scripts/fetch_tools.py --source ai-dir     # 指定源
"""
import argparse
import json
import re
import sys
import time
from datetime import datetime
from pathlib import Path
from urllib.parse import urlparse

import requests
from bs4 import BeautifulSoup

# 项目根目录
ROOT = Path(__file__).resolve().parent.parent
TOOLS_DIR = ROOT / "data" / "tools"
TOOLS_DIR.mkdir(parents=True, exist_ok=True)

# HTTP 头
HEADERS = {
    "User-Agent": "Mozilla/5.0 (compatible; AIToolsPricingBot/1.0; +https://aitoolspricing.com/bot)"
}

# 分类映射
CATEGORY_MAP = {
    "chat": ["chat", "assistant", "conversation", "gpt"],
    "writing": ["writing", "content", "copywriting", "blog"],
    "image": ["image", "art", "design", "midjourney", "dalle", "stable diffusion"],
    "video": ["video", "film", "animation"],
    "audio": ["audio", "music", "voice", "tts", "speech"],
    "code": ["code", "developer", "programming", "copilot"],
    "data": ["data", "analytics", "bi"],
    "search": ["search", "rag", "knowledge"],
    "agent": ["agent", "automation", "workflow"],
    "design": ["design", "ui", "ux", "logo"],
    "productivity": ["productivity", "notes", "calendar", "task"],
}


def map_category(text: str) -> str:
    text = text.lower()
    for cat, keywords in CATEGORY_MAP.items():
        if any(k in text for k in keywords):
            return cat
    return "other"


def slugify(name: str) -> str:
    s = re.sub(r"[^\w\s-]", "", name.lower())
    s = re.sub(r"[\s_-]+", "-", s).strip("-")
    return s


def fetch_ai_directory(limit: int = 50) -> list:
    """从 ai-directory.com 采集。"""
    url = "https://ai-directory.com"
    try:
        resp = requests.get(url, headers=HEADERS, timeout=30)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, "lxml")
        tools = []
        for card in soup.select(".tool-card, .ai-tool, article"):
            name_el = card.select_one("h2, h3, .name, .title")
            link_el = card.select_one("a[href]")
            desc_el = card.select_one(".description, .desc, p")
            if not name_el or not link_el:
                continue
            name = name_el.get_text(strip=True)
            href = link_el["href"]
            if href.startswith("/"):
                href = "https://ai-directory.com" + href
            desc = desc_el.get_text(strip=True) if desc_el else ""
            tools.append({
                "name": name,
                "url": href,
                "description": desc,
                "category": map_category(name + " " + desc),
                "source": "ai-directory",
            })
            if len(tools) >= limit:
                break
        return tools
    except Exception as e:
        print(f"[ai-directory] error: {e}")
        return []


def fetch_taaft(limit: int = 50) -> list:
    """从 theresanaiforthat.com 采集。"""
    url = "https://theresanaiforthat.com/ai-tools/"
    try:
        resp = requests.get(url, headers=HEADERS, timeout=30)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, "lxml")
        tools = []
        for card in soup.select("[class*='tool'], .ai-card, article"):
            name_el = card.select_one("h2, h3, .name")
            link_el = card.select_one("a[href*='/ai/']")
            desc_el = card.select_one(".desc, p, .description")
            if not name_el or not link_el:
                continue
            name = name_el.get_text(strip=True)
            href = link_el["href"]
            if href.startswith("/"):
                href = "https://theresanaiforthat.com" + href
            desc = desc_el.get_text(strip=True) if desc_el else ""
            tools.append({
                "name": name,
                "url": href,
                "description": desc,
                "category": map_category(name + " " + desc),
                "source": "taaft",
            })
            if len(tools) >= limit:
                break
        return tools
    except Exception as e:
        print(f"[taaft] error: {e}")
        return []


def fetch_producthunt(limit: int = 50) -> list:
    """从 ProductHunt RSS 采集(避免API鉴权)。"""
    url = "https://www.producthunt.com/feed"
    try:
        resp = requests.get(url, headers=HEADERS, timeout=30)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, "xml")
        tools = []
        for item in soup.find_all("item")[:limit]:
            title = item.find("title").get_text(strip=True) if item.find("title") else ""
            link = item.find("link").get_text(strip=True) if item.find("link") else ""
            desc = item.find("description").get_text(strip=True) if item.find("description") else ""
            desc_clean = BeautifulSoup(desc, "lxml").get_text(strip=True)[:300]
            if not title or not link:
                continue
            tools.append({
                "name": title.split(" — ")[0].split(" - ")[0].strip(),
                "url": link,
                "description": desc_clean,
                "category": map_category(title + " " + desc_clean),
                "source": "producthunt",
            })
        return tools
    except Exception as e:
        print(f"[producthunt] error: {e}")
        return []


def dedupe(tools: list) -> list:
    """按URL域名+工具名去重。"""
    seen = set()
    out = []
    for t in tools:
        domain = urlparse(t["url"]).netloc.lower().replace("www.", "")
        key = (domain, t["name"].lower())
        if key in seen:
            continue
        seen.add(key)
        out.append(t)
    return out


def tool_exists(slug: str) -> bool:
    return (TOOLS_DIR / f"{slug}.json").exists()


def save_tool(tool_data: dict) -> bool:
    """保存工具基础数据。如果已有则保留原数据，仅更新source。"""
    slug = slugify(tool_data["name"])
    if not slug:
        return False
    path = TOOLS_DIR / f"{slug}.json"
    if path.exists():
        existing = json.loads(path.read_text(encoding="utf-8"))
        existing.setdefault("sources", []).append({
            "source": tool_data.get("source"),
            "url": tool_data.get("url"),
            "fetched_at": datetime.utcnow().isoformat(),
        })
        existing["last_seen"] = datetime.utcnow().isoformat()
        path.write_text(json.dumps(existing, indent=2, ensure_ascii=False), encoding="utf-8")
        return False  # not new

    # 新工具
    record = {
        "id": slug,
        "slug": slug,
        "name": tool_data["name"],
        "vendor": {
            "name": tool_data["name"],
            "url": urlparse(tool_data["url"]).scheme + "://" + urlparse(tool_data["url"]).netloc,
        },
        "category": tool_data["category"],
        "usecases": [],
        "url": tool_data["url"],
        "affiliate_url": tool_data["url"],
        "tier": "unknown",
        "pricing": {
            "currency": "USD",
            "plans": [],
            "price_per_query": 0,
            "free_tier": {"available": False}
        },
        "features": [],
        "limits": [],
        "saving_tips": [],
        "alternatives": [],
        "rating": {
            "value_for_money": 3,
            "feature_richness": 3,
            "ease_of_use": 3,
            "overall": 3.0
        },
        "sources": [{
            "source": tool_data.get("source"),
            "url": tool_data.get("url"),
            "fetched_at": datetime.utcnow().isoformat(),
        }],
        "description": tool_data.get("description", ""),
        "first_seen": datetime.utcnow().isoformat(),
        "last_seen": datetime.utcnow().isoformat(),
        "last_updated": datetime.utcnow().isoformat(),
        "last_verified": None,
        "verified_by": None,
        "needs_review": True,  # 标记需人工/AI补全
    }
    path.write_text(json.dumps(record, indent=2, ensure_ascii=False), encoding="utf-8")
    return True


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--limit", type=int, default=50, help="每个源最多采集数量")
    ap.add_argument("--new-only", action="store_true", help="仅保存新工具")
    ap.add_argument("--source", choices=["ai-dir", "taaft", "producthunt", "all"], default="all")
    args = ap.parse_args()

    print(f"=== fetch_tools start (source={args.source}, limit={args.limit}) ===")
    all_tools = []

    if args.source in ("ai-dir", "all"):
        print("[1/3] fetching ai-directory...")
        all_tools.extend(fetch_ai_directory(args.limit))
        time.sleep(2)

    if args.source in ("taaft", "all"):
        print("[2/3] fetching TAAFT...")
        all_tools.extend(fetch_taaft(args.limit))
        time.sleep(2)

    if args.source in ("producthunt", "all"):
        print("[3/3] fetching ProductHunt...")
        all_tools.extend(fetch_producthunt(args.limit))

    print(f"\nTotal raw: {len(all_tools)}")
    unique = dedupe(all_tools)
    print(f"After dedupe: {len(unique)}")

    new_count = 0
    for t in unique:
        if save_tool(t):
            new_count += 1
            print(f"  ✨ NEW: {t['name']} ({t['url']})")
        else:
            if not args.new_only:
                print(f"  ✓ exists: {t['name']}")

    print(f"\n=== fetch_tools done. New: {new_count}, Total: {len(unique)} ===")


if __name__ == "__main__":
    main()

### 3.2 完整 `scripts/fetch_prices.py`

```python
"""
价格数据采集脚本。
1. 读取 data/tools/*.json 获取工具列表
2. 访问每个工具的厂商定价页
3. 用 BeautifulSoup 提取价格信息
4. 对难以解析的页面调用 Claude 推理
5. 与历史价格对比，标记 change_type
6. 更新 data/prices/{slug}.json
7. 若有变更，写入 .cache/price_changes.json 供告警

用法:
    python3 scripts/fetch_prices.py                    # 全部工具
    python3 scripts/fetch_prices.py --tool chatgpt-plus
    python3 scripts/fetch_prices.py --with-diff        # 启用diff
    python3 scripts/fetch_prices.py --with-diff --alert # 启用告警
"""
import argparse
import json
import os
import re
import sys
import time
from datetime import datetime, date
from pathlib import Path

import requests
from bs4 import BeautifulSoup

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))
from scripts.lib.ai_client import ClaudeClient
from scripts.lib.schema import validate_price

TOOLS_DIR = ROOT / "data" / "tools"
PRICES_DIR = ROOT / "data" / "prices"
PRICES_DIR.mkdir(parents=True, exist_ok=True)
CACHE_DIR = ROOT / ".cache"
CACHE_DIR.mkdir(exist_ok=True)

HEADERS = {
    "User-Agent": "Mozilla/5.0 (compatible; AIToolsPricingBot/1.0; +https://aitoolspricing.com/bot)"
}

# 已知厂商定价页(优先用)
KNOWN_PRICING_PAGES = {
    "chatgpt-plus": "https://openai.com/chatgpt/pricing/",
    "claude-pro": "https://www.anthropic.com/pricing",
    "gemini-advanced": "https://one.google.com/about/ai-premium",
    "midjourney": "https://docs.midjourney.com/docs/plans",
    "perplexity-pro": "https://www.perplexity.ai/pricing",
    "github-copilot": "https://github.com/features/copilot",
    "notion-ai": "https://www.notion.so/pricing",
    "jasper": "https://www.jasper.ai/pricing",
    "copy-ai": "https://www.copy.ai/pricing",
    "elevenlabs": "https://elevenlabs.io/pricing",
    "suno": "https://suno.com/pricing",
    "runway": "https://runwayml.com/pricing",
}


def fetch_html(url: str) -> str:
    """获取HTML文本。"""
    resp = requests.get(url, headers=HEADERS, timeout=30)
    resp.raise_for_status()
    return resp.text


def parse_price_text(text: str) -> dict:
    """从文本中正则提取价格信息。"""
    plans = []
    # 匹配 $20/month, $20/mo, $240/year 等
    pattern = re.compile(
        r"(?P<plan>[A-Z][a-zA-Z\s]{1,30}?)\s*[•\-\|:]?\s*\$(?P<price>\d+(?:\.\d{1,2})?)\s*/\s*(?P<cycle>month|mo|year|yr|month\b)",
        re.IGNORECASE,
    )
    for m in pattern.finditer(text):
        plan_name = m.group("plan").strip()
        price = float(m.group("price"))
        cycle = m.group("cycle").lower()
        cycle_norm = "monthly" if cycle.startswith("mo") else "annual"
        if plan_name and price >= 0:
            plans.append({
                "name": plan_name,
                "price": price,
                "billing_cycle": cycle_norm,
            })
    return plans


def extract_with_claude(html: str, tool_name: str) -> dict:
    """用 Claude 从 HTML 中提取价格信息。"""
    client = ClaudeClient()
    # 清理 HTML，只保留主体文本
    soup = BeautifulSoup(html, "lxml")
    for tag in soup(["script", "style", "noscript", "svg"]):
        tag.decompose()
    text = soup.get_text("\n", strip=True)[:8000]  # 截断

    prompt = f"""You are a precise data extraction assistant. Extract pricing information from the following pricing page text for the AI tool "{tool_name}".

Return JSON ONLY (no markdown, no explanation) in this exact schema:
{{
  "currency": "USD",
  "plans": [
    {{
      "name": "Free|Plus|Pro|Team|Enterprise|...",
      "price": 0,
      "billing_cycle": "monthly|annual|one_time|per_token",
      "annual_price": null,
      "description": "short description",
      "features": ["feature 1", "feature 2"],
      "limits": ["limit 1"]
    }}
  ],
  "free_tier": {{
    "available": true|false,
    "daily_limit": null,
    "monthly_limit": null,
    "features": []
  }}
}}

If you cannot find pricing info, return: {{"currency":"USD","plans":[],"free_tier":{{"available":false}}}}

Pricing page text:
---
{text}
---
"""
    result = client.complete(prompt, temperature=0, max_tokens=2048)
    # 容错: 去掉 markdown 包裹
    result = result.strip()
    if result.startswith("```"):
        result = re.sub(r"^```(?:json)?\s*", "", result)
        result = re.sub(r"\s*```$", "", result)
    try:
        return json.loads(result)
    except json.JSONDecodeError:
        print(f"[claude] failed to parse JSON for {tool_name}")
        return {"currency": "USD", "plans": [], "free_tier": {"available": False}}


def get_wayback_url(url: str) -> str:
    """获取 Wayback Machine 最近快照。"""
    api = f"https://archive.org/wayback/available?url={url}"
    try:
        resp = requests.get(api, timeout=15)
        data = resp.json()
        snapshot = data.get("archived_snapshots", {}).get("closest", {})
        if snapshot.get("available"):
            return snapshot.get("url")
    except Exception:
        pass
    return ""


def load_price_history(slug: str) -> dict:
    path = PRICES_DIR / f"{slug}.json"
    if path.exists():
        return json.loads(path.read_text(encoding="utf-8"))
    return {"tool_id": slug, "currency": "USD", "history": []}


def compute_change(old_price: float, new_price: float) -> tuple:
    if old_price is None:
        return ("initial", 0, 0)
    if new_price == old_price:
        return ("unchanged", 0, 0)
    delta = new_price - old_price
    pct = (delta / old_price * 100) if old_price > 0 else 0
    return ("increase" if delta > 0 else "decrease", round(delta, 2), round(pct, 2))


def fetch_tool_price(tool: dict, with_diff: bool) -> dict:
    """采集单工具价格。返回 {slug, changes: [], new_pricing: {}}"""
    slug = tool["slug"]
    pricing_url = KNOWN_PRICING_PAGES.get(slug) or tool.get("url", "")
    if not pricing_url:
        return {"slug": slug, "changes": [], "error": "no pricing url"}

    print(f"\n[{slug}] fetching {pricing_url}")
    try:
        html = fetch_html(pricing_url)
    except Exception as e:
        print(f"  ❌ fetch failed: {e}")
        return {"slug": slug, "changes": [], "error": str(e)}

    # 优先正则
    parsed = parse_price_text(html)
    if not parsed:
        # 回退到 Claude
        print(f"  regex empty, trying Claude...")
        parsed = extract_with_claude(html, tool["name"])

    if not parsed.get("plans"):
        print(f"  ⚠️ no plans found")
        return {"slug": slug, "changes": [], "error": "no plans"}

    # 加载历史
    history = load_price_history(slug)
    today = date.today().isoformat()
    changes = []

    for plan in parsed["plans"]:
        # 查找同 plan 同 cycle 的上次价格
        prev = None
        for h in reversed(history["history"]):
            if h["plan"] == plan["name"] and h.get("billing_cycle") == plan["billing_cycle"]:
                prev = h
                break
        old_price = prev["price"] if prev else None
        new_price = plan["price"]
        change_type, delta, pct = ("initial", 0, 0) if prev is None else compute_change(old_price, new_price)

        if with_diff and change_type in ("increase", "decrease", "new_plan", "initial"):
            changes.append({
                "plan": plan["name"],
                "old_price": old_price,
                "new_price": new_price,
                "change_type": change_type,
                "delta": delta,
                "delta_pct": pct,
            })

        # 追加历史
        history["history"].append({
            "date": today,
            "plan": plan["name"],
            "price": new_price,
            "billing_cycle": plan["billing_cycle"],
            "change_type": change_type,
            "delta": delta,
            "delta_pct": pct,
            "source_url": pricing_url,
            "source_snapshot": get_wayback_url(pricing_url),
        })

    # 更新工具主数据中的 pricing
    tool_path = TOOLS_DIR / f"{slug}.json"
    if tool_path.exists():
        tool_data = json.loads(tool_path.read_text(encoding="utf-8"))
        tool_data["pricing"] = parsed
        tool_data["last_verified"] = datetime.utcnow().isoformat()
        tool_data["verified_by"] = "fetch_prices.py + claude-3-5-sonnet"
        tool_data["needs_review"] = bool(changes)  # 有变更则需要复审
        tool_path.write_text(json.dumps(tool_data, indent=2, ensure_ascii=False), encoding="utf-8")

    # 校验并保存历史
    ok, errs = validate_price(history)
    if not ok:
        print(f"  ⚠️ schema validation failed: {errs}")
    (PRICES_DIR / f"{slug}.json").write_text(
        json.dumps(history, indent=2, ensure_ascii=False), encoding="utf-8"
    )
    print(f"  ✅ saved. changes: {len(changes)}")
    return {"slug": slug, "changes": changes, "new_pricing": parsed}


def send_alert(all_changes: list):
    """价格变更告警(Slack webhook)。"""
    webhook = os.getenv("SLACK_WEBHOOK")
    if not webhook or not all_changes:
        return

def split_frontmatter(md: str) -> tuple:
    """分离 frontmatter 与正文。"""
    if not md.startswith("---"):
        return "", md
    parts = md.split("---", 2)
    if len(parts) < 3:
        return "", md
    return parts[1] + "---", parts[2]


def translate_text(text: str, target_lang_name: str, content_type: str) -> str:
    """调用 GPT-4o 翻译正文。保留 Markdown 结构、URL、代码块、JSON-LD。"""
    client = GPT4oClient()

    system = f"""You are a professional translator specializing in AI tools and SaaS content. Translate from English to {target_lang_name}.

Strict rules:
- Preserve ALL Markdown formatting (headings, tables, lists, code blocks, links).
- Do NOT translate URLs, code, JSON-LD blocks, Hugo shortcodes, frontmatter variables.
- Translate ONLY natural language text.
- Keep brand names in original (ChatGPT, Claude, Gemini, Midjourney, OpenAI, Anthropic).
- Keep technical terms in English if no widely accepted translation (API, LLM, GPT-4o).
- Adapt currency notation naturally (e.g., "$20/月" for Chinese).
- For RTL languages (Arabic), keep Markdown structure but flow text RTL.

Translated Markdown:"""

def translate_file(src_path: Path, target_lang: str) -> bool:
    """翻译单个 Markdown 文件。"""
    if not src_path.exists():
        return False

    target_code, target_name = LANG_MAP[target_lang]
    rel_path = src_path.relative_to(EN_CONTENT)
    dst_path = ROOT / "content" / target_code / rel_path
    dst_path.parent.mkdir(parents=True, exist_ok=True)

    if dst_path.exists():
        # 已存在,跳过
        return False

    md = src_path.read_text(encoding="utf-8")
    frontmatter, body = split_frontmatter(md)

    if not body.strip():
        return False

    content_type = src_path.parent.name  # tool, comparison, guide
    translated_body = translate_text(body, target_name, content_type)

    # 更新 frontmatter 中的 slug/date 等可保留,仅修改 lastmod
    final = "---" + frontmatter + "\n\n" + translated_body
    dst_path.write_text(final, encoding="utf-8")
    return True


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--target-langs", required=True, help="逗号分隔的目标语言")
    ap.add_argument("--tool", help="仅翻译指定工具相关页面")

    print(f"  source files: {len(src_files)}")
    print(f"  target langs: {targets}")

    total = 0
    for src in src_files:
        for lang in targets:
            try:
                if translate_file(src, lang):
                    total += 1
                    print(f"  ✅ {src.relative_to(EN_CONTENT)} → {lang}")
            except Exception as e:
                print(f"  ❌ {src.relative_to(EN_CONTENT)} → {lang}: {e}")

    print(f"\n=== done. translated: {total} files ===")


if __name__ == "__main__":
    main()

# 用 Gemini 文本生成 SVG (Gemini 1.5 Flash 免费但无原生图生成 API,用文本生成SVG代码)

def generate_comparison_og_image(comparison_slug: str, tools: list) -> str:
    """生成对比页 OG 图(SVG)。"""
    client = GeminiClient()
    tool_names = " vs ".join(t["name"] for t in tools)
    prices = " | ".join(f"${t['pricing']['plans'][1]['price']}/mo" if len(t["pricing"]["plans"]) > 1 else "N/A" for t in tools)

    prompt = f"""Generate an SVG image (1200x630, OG image size) for a comparison article.

Title: "{tool_names} Pricing Comparison"
Subtitle: "{prices}"
Brand: "AI Tools Pricing"
Year: "2026"

Design requirements:
- Modern, clean design
- Two-color palette: #2563eb (blue) and #f59e0b (orange)
- White background
- Large bold title centered
- Subtitle below title in smaller font
- Brand name in top-left, year in top-right
- A simple icon or chart element representing comparison
- Output SVG code ONLY, no markdown, no explanation.

SVG:"""
    svg = client.complete(prompt, max_tokens=2048, temperature=0.4)
    # 清理
    svg = svg.strip()
    if svg.startswith("```"):
        svg = svg.split("\n", 1)[1] if "\n" in svg else svg[3:]
    if svg.endswith("```"):
        svg = svg.rsplit("```", 1)[0]
    if not svg.startswith("<svg"):
        svg = "<svg " + svg
    return svg


def generate_price_chart_svg(tool: dict) -> str:
    """生成单工具价格历史 SVG 图。"""
    client = GeminiClient()
    prices_path = ROOT / "data" / "prices" / f"{tool['slug']}.json"
    if not prices_path.exists():

    history = json.loads(prices_path.read_text(encoding="utf-8"))["history"]

    prompt = f"""Generate an SVG line chart (800x400) showing price history for {tool['name']}.

Price history data (JSON):

Design requirements:
- Clean, minimal line chart
- X-axis: dates
- Y-axis: price in USD
- Single blue line (#2563eb), 2px stroke
- Data points marked with circles
- Labels: title "{tool['name']} Price History", axis labels
- White background, light gray grid
- Output SVG code ONLY, no markdown.

SVG:"""
    svg = client.complete(prompt, max_tokens=2048, temperature=0.3)
    return svg.strip()

## 8. 内容审核

### 8.1 完整 `scripts/audit_content.py`

```python
"""
内容审核脚本。
对 AI 生成的对比/攻略页进行审核:
1. 价格准确性: 与 data/tools/*.json 中的价格比对
2. 合规性: 联盟披露、价格免责、商标使用
3. 链接有效性: 内部链接是否存在
4. 内容质量: 字数、结构、重复度

用法:
    python3 scripts/audit_content.py --all
    python3 scripts/audit_content.py --file content/en/comparison/chatgpt-vs-claude-pricing.md
    python3 scripts/audit_content.py --new-only
"""
import argparse
import json
import re
import sys
from datetime import datetime
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))
from scripts.lib.ai_client import ClaudeClient

CONTENT_DIR = ROOT / "content"
TOOLS_DIR = ROOT / "data" / "tools"


def audit_prices(md: str, slug_hints: list) -> dict:
    """检查 Markdown 中的价格是否与工具数据一致。"""
    issues = []
    # 提取所有 $X 形式价格
    prices_in_md = re.findall(r"\$(\d+(?:\.\d{1,2})?)", md)
    expected = set()
    for slug in slug_hints:
        path = TOOLS_DIR / f"{slug}.json"
        if not path.exists():
            continue
        tool = json.loads(path.read_text(encoding="utf-8"))
        for plan in tool.get("pricing", {}).get("plans", []):
            if plan["price"] > 0:
                expected.add(str(plan["price"]))
            if plan.get("annual_price"):
                expected.add(str(plan["annual_price"]))

    found = set(prices_in_md)
    # 不在 expected 中的可疑价格(可能是捏造)
    suspicious = found - expected - {"0", "1", "5", "10"}  # 允许常见小数字
    if suspicious:
        issues.append({
            "type": "price_mismatch",
            "severity": "high",
            "msg": f"Prices in content not in tool data: {suspicious}"
        })
    return {"issues": issues, "prices_found": list(found)}


def audit_compliance(md: str) -> dict:
    """合规性审核。"""
    issues = []
    lower = md.lower()
    if "affiliate" not in lower and "disclosure" not in lower:
        issues.append({"type": "missing_affiliate_disclosure", "severity": "high", "msg": "No affiliate disclosure found"})
    if "disclaimer" not in lower and "reference only" not in lower:
        issues.append({"type": "missing_price_disclaimer", "severity": "high", "msg": "No price disclaimer found"})
    # 商标合规: 不应有"OpenAI ChatGPT"作为我们产品
    if re.search(r"\bour\s+(?:ChatGPT|Claude|Gemini|Midjourney)\b", md, re.I):
        issues.append({"type": "trademark_misuse", "severity": "high", "msg": "Trademark used as own product"})
    return {"issues": issues}


def audit_links(md: str) -> dict:
    """内部链接审核。"""
    issues = []
    links = re.findall(r"\[([^\]]+)\]\((/[^)]+)\)", md)
    for text, href in links:
        # 检查对应文件是否存在
        local_path = ROOT / "content" / "en" / href.strip("/").replace("/pricing/", "/pricing.md").replace("/", "/")
        # 简化检查: 仅看 slug 是否在 data/tools 中
        slug_match = re.search(r"/tool/([a-z0-9-]+)/", href)
        if slug_match:
            slug = slug_match.group(1)
            if not (TOOLS_DIR / f"{slug}.json").exists():
                issues.append({"type": "broken_link", "severity": "medium", "msg": f"Link to unknown tool: {href}"})
    return {"issues": issues, "link_count": len(links)}


def audit_quality(md: str) -> dict:
    """内容质量审核。"""
    issues = []
    word_count = len(md.split())
    if word_count < 800:
        issues.append({"type": "too_short", "severity": "medium", "msg": f"Word count {word_count} < 800"})
    if word_count > 5000:
        issues.append({"type": "too_long", "severity": "low", "msg": f"Word count {word_count} > 5000"})
    # 必备章节
    required_sections = ["TL;DR", "FAQ", "Verdict", "Price", "Features"]
    for sec in required_sections:
        if sec.lower() not in md.lower():
            issues.append({"type": "missing_section", "severity": "medium", "msg": f"Missing section: {sec}"})
    return {"issues": issues, "word_count": word_count}


def audit_with_claude(md: str, content_type: str) -> dict:
    """Claude 深度审核。"""
    client = ClaudeClient()
    prompt = f"""Audit the following AI-generated {content_type} article for an AI tools pricing comparison website.

Check for:
1. Factual errors (especially wrong prices, wrong feature claims)
2. Bias toward any tool without evidence
3. Misleading claims
4. Missing FTC affiliate disclosure
5. Missing price disclaimer
6. Trademark misuse
7. Hallucinated features that don't exist
8. Outdated information
9. Tone issues (too promotional, too negative)
10. SEO issues (missing keywords, thin content)

Return JSON ONLY in this format:
{{
  "pass": true|false,
  "score": 0-100,
  "issues": [
    {{"severity": "high|medium|low", "category": "...", "msg": "..."}}
  ],
  "suggestions": ["..."]
}}

Article to audit:
---
{md[:6000]}
---
"""
    result = client.complete(prompt, temperature=0, max_tokens=2048)
    try:
        return json.loads(result.strip().strip("`"))
    except json.JSONDecodeError:
        return {"pass": False, "score": 0, "issues": [{"severity": "high", "category": "parse", "msg": "Claude output not JSON"}], "suggestions": []}


def audit_file(path: Path) -> dict:
    """审核单文件。"""
    md = path.read_text(encoding="utf-8")
    # 提取 tool slugs
    slug_hints = re.findall(r"tool_id:\s*[\"']([a-z0-9-]+)[\"']", md)
    slug_hints += re.findall(r"/tool/([a-z0-9-]+)/", md)
    slug_hints = list(set(slug_hints))

    result = {
        "file": str(path.relative_to(ROOT)),
        "audited_at": datetime.utcnow().isoformat(),
        "checks": {
            "prices": audit_prices(md, slug_hints),
            "compliance": audit_compliance(md),
            "links": audit_links(md),
            "quality": audit_quality(md),
        }
    }

    # Claude 深度审核
    content_type = path.parent.name
    result["ai_audit"] = audit_with_claude(md, content_type)

    all_issues = []
    for v in result["checks"].values():
        all_issues.extend(v.get("issues", []))
    all_issues.extend(result["ai_audit"].get("issues", []))
    result["total_issues"] = len(all_issues)
    result["high_severity"] = sum(1 for i in all_issues if i.get("severity") == "high")
    result["pass"] = result["high_severity"] == 0 and result["ai_audit"].get("pass", False)
    return result


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--all", action="store_true")
    ap.add_argument("--file")
    ap.add_argument("--new-only", action="store_true", help="仅审核今日新增")
    args = ap.parse_args()

    files = []
    if args.file:
        files = [ROOT / args.file]
    elif args.all:
        files = list((CONTENT_DIR / "en").rglob("*.md"))
    else:
        print("Use --all or --file path")
        return

    print(f"=== audit_content start. files: {len(files)} ===")
    results = []
    pass_count = 0
    for f in files:
        if not f.exists():
            continue
        print(f"\n[{f.relative_to(ROOT)}]")
        try:
            r = audit_file(f)
            results.append(r)
            status = "✅ PASS" if r["pass"] else "❌ FAIL"
            print(f"  {status} (issues: {r['total_issues']}, high: {r['high_severity']})")
            for issue in (r["checks"]["prices"]["issues"] + r["checks"]["compliance"]["issues"] + r["checks"]["links"]["issues"] + r["checks"]["quality"]["issues"] + r["ai_audit"].get("issues", [])):
                print(f"    - [{issue.get('severity','?')}] {issue.get('msg', issue)}")
            if r["pass"]:
                pass_count += 1
        except Exception as e:
            print(f"  ❌ error: {e}")

    # 保存审核报告
    audit_path = ROOT / ".cache" / "audit_report.json"
    audit_path.parent.mkdir(parents=True, exist_ok=True)
    audit_path.write_text(json.dumps(results, indent=2, ensure_ascii=False), encoding="utf-8")
    print(f"\n=== audit done. pass: {pass_count}/{len(files)} ===")
    print(f"📁 report: {audit_path}")


if __name__ == "__main__":
    main()
```

## 13. 异常处理与降级方案

### 13.1 异常分类与处理

| 异常 | 检测 | 处理 | 降级 |
|------|------|------|------|
| 厂商定价页改版 | fetch_prices解析0个plan | 触发Claude推理 | Claude也失败→标记needs_review |
| Claude API 限流 | 429错误 | 指数退避重试3次 | 失败→缓存上次结果 |
| GPT-4o 翻译失败 | timeout/错误 | 重试3次 | 失败→用GPT-4o-mini |
| Gemini 配图失败 | 错误 | 跳过配图 | 用默认OG图 |
| 内容审核 fail | high severity issue | 不发布 | 触发Claude重写 |
| Schema校验 fail | 数据不符schema | 不入库 | 报错+人工修 |
| GitHub Actions 失败 | exit code非0 | 重试1次 | Slack告警+人工 |
| Cloudflare 部署失败 | 部署状态failed | 回滚到上次成功 | Slack告警 |
| 价格采集 0 工具 | 全部失败 | 检查网络 | 保留昨日数据 |
| Wayback Machine 不可用 | API错误 | 跳过snapshot | 仅保存source_url |
